#include <stdio.h>

int main() {

  int n,i;

  printf("\n\nEnter a number: " );
  scanf("%d",&n );

  for (i=1;i<=10;i++) {

    printf("%d x %d = %d \n",n,i,n*i );
  }

  return 0;
}
